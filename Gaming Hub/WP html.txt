    <script>
        // Global game state
        let currentGame = null;

        // Menu navigation
        function showMenu() {
            document.getElementById('menu').style.display = 'grid';
            document.querySelectorAll('.game-area').forEach(area => area.classList.remove('active'));
            currentGame = null;
        }

        function showGame(gameId) {
            document.getElementById('menu').style.display = 'none';
            document.getElementById(gameId).classList.add('active');
            currentGame = gameId;
            
            // Initialize games when shown
            if (gameId === 'ticTacToe') initTicTacToe();
            if (gameId === 'memory') initMemory();
            if (gameId === 'numberGuess') initNumberGuess();
        }

        // Tic Tac Toe Game
        let ticTacToeGame = {
            board: Array(9).fill(''),
            currentPlayer: 'X',
            gameOver: false
        };

        function initTicTacToe() {
            const board = document.getElementById('ticTacToeBoard');
            board.innerHTML = '';
            ticTacToeGame = { board: Array(9).fill(''), currentPlayer: 'X', gameOver: false };
            
            for (let i = 0; i < 9; i++) {
                const cell = document.createElement('button');
                cell.className = 'tic-tac-toe-cell';
                cell.onclick = () => playTicTacToe(i);
                board.appendChild(cell);
            }
            document.getElementById('ticTacToeStatus').textContent = "Player X's turn";
        }

        function playTicTacToe(index) {
            if (ticTacToeGame.board[index] || ticTacToeGame.gameOver) return;
            
            ticTacToeGame.board[index] = ticTacToeGame.currentPlayer;
            document.getElementById('ticTacToeBoard').children[index].textContent = ticTacToeGame.currentPlayer;
            
            if (checkTicTacToeWinner()) {
                document.getElementById('ticTacToeStatus').textContent = `Player ${ticTacToeGame.currentPlayer} wins!`;
                ticTacToeGame.gameOver = true;
            } else if (ticTacToeGame.board.every(cell => cell)) {
                document.getElementById('ticTacToeStatus').textContent = "It's a tie!";
                ticTacToeGame.gameOver = true;
            } else {
                ticTacToeGame.currentPlayer = ticTacToeGame.currentPlayer === 'X' ? 'O' : 'X';
                document.getElementById('ticTacToeStatus').textContent = `Player ${ticTacToeGame.currentPlayer}'s turn`;
            }
        }

        function checkTicTacToeWinner() {
            const winPatterns = [
                [0,1,2], [3,4,5], [6,7,8],
                [0,3,6], [1,4,7], [2,5,8],
                [0,4,8], [2,4,6]
            ];
            
            return winPatterns.some(pattern => {
                const [a, b, c] = pattern;
                return ticTacToeGame.board[a] && 
                       ticTacToeGame.board[a] === ticTacToeGame.board[b] && 
                       ticTacToeGame.board[a] === ticTacToeGame.board[c];
            });
        }

        function resetTicTacToe() {
            initTicTacToe();
        }

        // Rock Paper Scissors Game
        let rpsScore = { player: 0, computer: 0 };

        function playRPS(playerChoice) {
            const choices = ['rock', 'paper', 'scissors'];
            const computerChoice = choices[Math.floor(Math.random() * 3)];
            
            let result = '';
            if (playerChoice === computerChoice) {
                result = "It's a tie!";
            } else if (
                (playerChoice === 'rock' && computerChoice === 'scissors') ||
                (playerChoice === 'paper' && computerChoice === 'rock') ||
                (playerChoice === 'scissors' && computerChoice === 'paper')
            ) {
                result = "You win!";
                rpsScore.player++;
            } else {
                result = "Computer wins!";
                rpsScore.computer++;
            }
            
            document.getElementById('rpsStatus').textContent = 
                `You: ${playerChoice}, Computer: ${computerChoice}. ${result}`;
            document.getElementById('rpsScore').textContent = 
                `You: ${rpsScore.player} | Computer: ${rpsScore.computer}`;
        }

        // Snake Game
        let snakeGame = {
            canvas: null,
            ctx: null,
            snake: [{x: 200, y: 200}],
            food: {x: 0, y: 0},
            dx: 20,
            dy: 0,
            score: 0,
            gameRunning: false
        };

        function startSnake() {
            snakeGame.canvas = document.getElementById('snakeCanvas');
            snakeGame.ctx = snakeGame.canvas.getContext('2d');
            snakeGame.snake = [{x: 200, y: 200}];
            snakeGame.dx = 20;
            snakeGame.dy = 0;
            snakeGame.score = 0;
            snakeGame.gameRunning = true;
            
            generateFood();
            document.addEventListener('keydown', changeDirection);
            gameLoop();
        }

        function gameLoop() {
            if (!snakeGame.gameRunning) return;
            
            setTimeout(() => {
                clearCanvas();
                moveSnake();
                drawSnake();
                drawFood();
                
                if (checkCollision()) {
                    snakeGame.gameRunning = false;
                    document.getElementById('snakeStatus').textContent = 'Game Over! Press Start Game to play again.';
                    return;
                }
                
                gameLoop();
            }, 100);
        }

        function clearCanvas() {
            snakeGame.ctx.fillStyle = 'rgba(0,0,0,0.3)';
            snakeGame.ctx.fillRect(0, 0, snakeGame.canvas.width, snakeGame.canvas.height);
        }

        function moveSnake() {
            const head = {x: snakeGame.snake[0].x + snakeGame.dx, y: snakeGame.snake[0].y + snakeGame.dy};
            snakeGame.snake.unshift(head);
            
            if (head.x === snakeGame.food.x && head.y === snakeGame.food.y) {
                snakeGame.score++;
                document.getElementById('snakeScore').textContent = `Score: ${snakeGame.score}`;
                generateFood();
            } else {
                snakeGame.snake.pop();
            }
        }

        function drawSnake() {
            snakeGame.ctx.fillStyle = '#4CAF50';
            snakeGame.snake.forEach(segment => {
                snakeGame.ctx.fillRect(segment.x, segment.y, 18, 18);
            });
        }

        function drawFood() {
            snakeGame.ctx.fillStyle = '#FF5722';
            snakeGame.ctx.fillRect(snakeGame.food.x, snakeGame.food.y, 18, 18);
        }

        function generateFood() {
            snakeGame.food.x = Math.floor(Math.random() * 20) * 20;
            snakeGame.food.y = Math.floor(Math.random() * 20) * 20;
        }

        function changeDirection(event) {
            const keyPressed = event.keyCode;
            const goingUp = snakeGame.dy === -20;
            const goingDown = snakeGame.dy === 20;
            const goingRight = snakeGame.dx === 20;
            const goingLeft = snakeGame.dx === -20;
            
            if (keyPressed === 37 && !goingRight) {
                snakeGame.dx = -20; snakeGame.dy = 0;
            }
            if (keyPressed === 38 && !goingDown) {
                snakeGame.dx = 0; snakeGame.dy = -20;
            }
            if (keyPressed === 39 && !goingLeft) {
                snakeGame.dx = 20; snakeGame.dy = 0;
            }
            if (keyPressed === 40 && !goingUp) {
                snakeGame.dx = 0; snakeGame.dy = 20;
            }
        }

        function checkCollision() {
            const head = snakeGame.snake[0];
            
            // Wall collision
            if (head.x < 0 || head.x >= 400 || head.y < 0 || head.y >= 400) {
                return true;
            }
            
            // Self collision
            for (let i = 1; i < snakeGame.snake.length; i++) {
                if (head.x === snakeGame.snake[i].x && head.y === snakeGame.snake[i].y) {
                    return true;
                }
            }
            
            return false;
        }

        // Memory Match Game
        let memoryGame = {
            cards: [],
            flippedCards: [],
            matchedPairs: 0,
            moves: 0,
            symbols: ['üçé', 'üçå', 'üçä', 'üçá', 'üçì', 'ü•ù', 'üçë', 'üçç']
        };

        function initMemory() {
            const board = document.getElementById('memoryBoard');
            board.innerHTML = '';
            memoryGame.cards = [...memoryGame.symbols, ...memoryGame.symbols]
                .sort(() => Math.random() - 0.5);
            memoryGame.flippedCards = [];
            memoryGame.matchedPairs = 0;
            memoryGame.moves = 0;
            
            memoryGame.cards.forEach((symbol, index) => {
                const card = document.createElement('button');
                card.className = 'memory-card';
                card.textContent = '?';
                card.onclick = () => flipCard(index);
                board.appendChild(card);
            });
            
            updateMemoryScore();
        }

        function flipCard(index) {
            if (memoryGame.flippedCards.length === 2) return;
            if (memoryGame.flippedCards.includes(index)) return;
            
            const cards = document.getElementById('memoryBoard').children;
            cards[index].textContent = memoryGame.cards[index];
            cards[index].classList.add('flipped');
            memoryGame.flippedCards.push(index);
            
            if (memoryGame.flippedCards.length === 2) {
                memoryGame.moves++;
                setTimeout(checkMemoryMatch, 1000);
            }
        }

        function checkMemoryMatch() {
            const [first, second] = memoryGame.flippedCards;
            const cards = document.getElementById('memoryBoard').children;
            
            if (memoryGame.cards[first] === memoryGame.cards[second]) {
                memoryGame.matchedPairs++;
                if (memoryGame.matchedPairs === 8) {
                    alert(`Congratulations! You completed the game in ${memoryGame.moves} moves!`);
                }
            } else {
                cards[first].textContent = '?';
                cards[second].textContent = '?';
                cards[first].classList.remove('flipped');
                cards[second].classList.remove('flipped');
            }
            
            memoryGame.flippedCards = [];
            updateMemoryScore();
        }

        function updateMemoryScore() {
            document.getElementById('memoryScore').textContent = 
                `Moves: ${memoryGame.moves} | Matches: ${memoryGame.matchedPairs}/8`;
        }

        function resetMemory() {
            initMemory();
        }

        // Number Guessing Game
        let numberGame = {
            targetNumber: 0,
            attempts: 0,
            gameOver: false
        };

        function initNumberGuess() {
            numberGame.targetNumber = Math.floor(Math.random() * 100) + 1;
            numberGame.attempts = 0;
            numberGame.gameOver = false;
            document.getElementById('guessStatus').textContent = 'Guess a number between 1 and 100!';
            document.getElementById('guessScore').textContent = 'Attempts: 0';
            document.getElementById('guessInput').value = '';
        }

        function makeGuess() {
            if (numberGame.gameOver) return;
            
            const guess = parseInt(document.getElementById('guessInput').value);
            if (!guess || guess < 1 || guess > 100) {
                document.getElementById('guessStatus').textContent = 'Please enter a valid number between 1 and 100!';
                return;
            }
            
            numberGame.attempts++;
            document.getElementById('guessScore').textContent = `Attempts: ${numberGame.attempts}`;
            
            if (guess === numberGame.targetNumber) {
                document.getElementById('guessStatus').textContent = 
                    `üéâ Congratulations! You guessed it in ${numberGame.attempts} attempts!`;
                numberGame.gameOver = true;
            } else if (guess < numberGame.targetNumber) {
                document.getElementById('guessStatus').textContent = 'üìà Too low! Try a higher number.';
            } else {
                document.getElementById('guessStatus').textContent = 'üìâ Too high! Try a lower number.';
            }
            
            document.getElementById('guessInput').value = '';
        }

        function resetNumberGuess() {
            initNumberGuess();
        }

        // Flappy Bird Game
        let flappyGame = {
            canvas: null,
            ctx: null,
            bird: {x: 50, y: 300, velocity: 0},
            pipes: [],
            score: 0,
            gameRunning: false
        };

        function startFlappy() {
            flappyGame.canvas = document.getElementById('flappyCanvas');
            flappyGame.ctx = flappyGame.canvas.getContext('2d');
            flappyGame.bird = {x: 50, y: 300, velocity: 0};
            flappyGame.pipes = [];
            flappyGame.score = 0;
            flappyGame.gameRunning = true;
            
            flappyGame.canvas.onclick = flap;
            flappyGameLoop();
        }

        function flappyGameLoop() {
            if (!flappyGame.gameRunning) return;
            
            // Clear canvas
            flappyGame.ctx.fillStyle = '#87CEEB';
            flappyGame.ctx.fillRect(0, 0, flappyGame.canvas.width, flappyGame.canvas.height);
            
            // Update bird
            flappyGame.bird.velocity += 0.5;
            flappyGame.bird.y += flappyGame.bird.velocity;
            
            // Draw bird
            flappyGame.ctx.fillStyle = '#FFD700';
            flappyGame.ctx.fillRect(flappyGame.bird.x, flappyGame.bird.y, 20, 20);
            
            // Generate pipes
            if (flappyGame.pipes.length === 0 || flappyGame.pipes[flappyGame.pipes.length - 1].x < 200) {
                const pipeHeight = Math.random() * 200 + 100;
                flappyGame.pipes.push({
                    x: flappyGame.canvas.width,
                    topHeight: pipeHeight,
                    bottomY: pipeHeight + 150
                });
            }
            
            // Update and draw pipes
            flappyGame.ctx.fillStyle = '#32CD32';
            flappyGame.pipes.forEach((pipe, index) => {
                pipe.x -= 2;
                
                // Draw top pipe
                flappyGame.ctx.fillRect(pipe.x, 0, 50, pipe.topHeight);
                // Draw bottom pipe
                flappyGame.ctx.fillRect(pipe.x, pipe.bottomY, 50, flappyGame.canvas.height - pipe.bottomY);
                
                // Check collision
                if (flappyGame.bird.x + 20 > pipe.x && flappyGame.bird.x < pipe.x + 50) {
                    if (flappyGame.bird.y < pipe.topHeight || flappyGame.bird.y + 20 > pipe.bottomY) {
                        endFlappyGame();
                        return;
                    }
                }
                
                // Score point
                if (pipe.x + 50 < flappyGame.bird.x && !pipe.scored) {
                    flappyGame.score++;
                    pipe.scored = true;
                    document.getElementById('flappyScore').textContent = `Score: ${flappyGame.score}`;
                }
                
                // Remove off-screen pipes
                if (pipe.x + 50 < 0) {
                    flappyGame.pipes.splice(index, 1);
                }
            });
            
            // Check ground/ceiling collision
            if (flappyGame.bird.y > flappyGame.canvas.height || flappyGame.bird.y < 0) {
                endFlappyGame();
                return;
            }
            
            requestAnimationFrame(flappyGameLoop);
        }

        function flap() {
            if (!flappyGame.gameRunning) return;
            flappyGame.bird.velocity = -8;
        }

        function endFlappyGame() {
            flappyGame.gameRunning = false;
            flappyGame.canvas.onclick = null;
            document.getElementById('flappyStatus').textContent = 
                `Game Over! Final Score: ${flappyGame.score}. Press Start Game to play again.`;
        }

        // Initialize on load
        document.getElementById('guessInput').addEventListener('keypress', function(e) {
            if (e.key === 'Enter') makeGuess();
        });
